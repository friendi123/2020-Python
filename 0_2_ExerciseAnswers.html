from scipy import stats
import numpy as np
# plot normal pdf
x = np.linspace(0,10,100)
y = stats.norm(5,3).pdf(x)

import matplotlib.pyplot as plt

plt.plot(x,y)
plt.show()

# plot normal cdf
 
x = np.linspace(0,10,100)
y = stats.norm(5,3).cdf(x)

plt.plot(x,y)
plt.show()
